```
let arr1 = []
for (let i = 0; i < 3; i++) {
    arr1[i] = function() {
        console.log(i)
    }
}
arr1[2]()

let arr2 = []
for (var j = 0; j < 3; j++) {
    arr2[j] = function() {
        console.log(j)
    }
}
arr2[2]()
```
